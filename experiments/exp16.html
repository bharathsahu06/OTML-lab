import numpy as np from sklearn.linear_model import Lasso, Ridge, ElasticNet # Sample data X = np.array([[1], [2], [3], [4]]) y = np.array([2, 4, 6, 8]) # L1 Regularization (Lasso) lasso = Lasso(alpha=0.1) lasso.fit(X, y) # L2 Regularization (Ridge) ridge = Ridge(alpha=0.1) ridge.fit(X, y) # Elastic Net elastic_net = ElasticNet(alpha=0.05, l1_ratio=0.5) elastic_net.fit(X, y) # Results print("L1 (Lasso):") print(f" Coefficient: {lasso.coef_[0]:.4f}") print(f" Intercept: {lasso.intercept_:.4f}") print("\nL2 (Ridge):") print(f" Coefficient: {ridge.coef_[0]:.4f}") print(f" Intercept: {ridge.intercept_:.4f}") print("\nElastic Net:") print(f" Coefficient: {elastic_net.coef_[0]:.4f}") print(f" Intercept: {elastic_net.intercept_:.4f}")
