import numpy as np # Dataset: (x1, x2, y) X = np.array([[1, 2], [2, 1], [3, 4]]) y = np.array([5, 6, 7]) # Initialize weights w = np.array([0.0, 0.0]) learning_rate = 0.01 lambda_reg = 0.1 # L1 regularization strength # Subgradient descent for epoch in range(100): # Predictions y_pred = X @ w # Error error = y_pred - y # Gradient grad = 2 * X.T @ error # Sub-gradient (add L1 penalty) sub_grad = grad + lambda_reg * np.sign(w) # Update w = w - learning_rate * sub_grad if epoch % 20 == 0: loss = np.mean(error**2) + lambda_reg * np.sum(np.abs(w)) print(f"Epoch {epoch}: Loss = {loss:.4f}, w = {w}") print(f"\nFinal w = {w}")
