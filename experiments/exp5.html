import cvxpy as cp # Define variables x = cp.Variable() y = cp.Variable() # Define objective: minimize 3x + 2y objective = cp.Minimize(3*x + 2*y) # Define constraints constraints = [ x + y >= 4, # x + y >= 4 x >= 0, # x >= 0 y >= 0 # y >= 0 ] # Create and solve problem = cp.Problem(objective, constraints) problem.solve() # Results print(f"Optimal x = {x.value:.6f}") print(f"Optimal y = {y.value:.6f}") print(f"Minimum value = {objective.value:.6f}") print(f"Status: {problem.status}")
